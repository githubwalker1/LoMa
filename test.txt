#导出模型 
支持这些导出模式
--kind detector        # 只导出 loma_detector.pt
--kind descriptor      # 只导出 loma_descriptor.pt
--kind matcher         # 只导出 loma_matcher.pt
--kind split           # 导出三段：detector + descriptor + matcher
--kind detect_describe # 保留旧逻辑：detector+descriptor 合在一起
--kind all             # 三段 + 旧的 detect_describe 都导出
